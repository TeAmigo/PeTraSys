047    package org.jfree.chart.annotations;
048    
049    import java.awt.Graphics2D;
050    import java.awt.geom.Rectangle2D;
051    
052    import org.jfree.chart.axis.ValueAxis;
053    import org.jfree.chart.plot.PlotRenderingInfo;
054    import org.jfree.chart.plot.XYPlot;
055    
056    /**
057     * The interface that must be supported by annotations that are to be added to
058     * an {@link XYPlot}.
059     */
060    public interface XYAnnotation {
061    
062        /**
063         * Draws the annotation.
064         *
065         * @param g2  the graphics device.
066         * @param plot  the plot.
067         * @param dataArea  the data area.
068         * @param domainAxis  the domain axis.
069         * @param rangeAxis  the range axis.
070         * @param rendererIndex  the renderer index.
071         * @param info  an optional info object that will be populated with
072         *              entity information.
073         */
074        public void draw(Graphics2D g2, XYPlot plot, Rectangle2D dataArea,
075                         ValueAxis domainAxis, ValueAxis rangeAxis,
076                         int rendererIndex,
077                         PlotRenderingInfo info);
078    
079    }

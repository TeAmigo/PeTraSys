044    package org.jfree.data.function;
045    
046    /**
047     * A normal distribution function.  See
048     * http://en.wikipedia.org/wiki/Normal_distribution.
049     */
050    public class NormalDistributionFunction2D implements Function2D {
051    
052        /** The mean. */
053        private double mean;
054    
055        /** The standard deviation. */
056        private double std;
057    
058        /** Precomputed factor for the function value. */
059        private double factor;
060    
061        /** Precomputed denominator for the function value. */
062        private double denominator;
063    
064        /**
065         * Constructs a new normal distribution function.
066         *
067         * @param mean  the mean.
068         * @param std  the standard deviation (> 0).
069         */
070        public NormalDistributionFunction2D(double mean, double std) {
071            if (std <= 0) {
072                throw new IllegalArgumentException("Requires 'std' > 0.");
073            }
074            this.mean = mean;
075            this.std = std;
076            // calculate constant values
077            this.factor = 1 / (std * Math.sqrt(2.0 * Math.PI));
078            this.denominator = 2 * std * std;
079        }
080    
081        /**
082         * Returns the mean for the function.
083         *
084         * @return The mean.
085         */
086        public double getMean() {
087            return this.mean;
088        }
089        
090        /**
091         * Returns the standard deviation for the function.
092         *
093         * @return The standard deviation.
094         */
095        public double getStandardDeviation() {
096            return this.std;
097        }
098    
099        /**
100         * Returns the function value.
101         *
102         * @param x  the x-value.
103         *
104         * @return The value.
105         */
106        public double getValue(double x) {
107            double z = x - this.mean;
108            return this.factor * Math.exp(-z * z / this.denominator);
109        }
110    
111    }
